SQL> CREATE OR REPLACE PROCEDURE prc_get_subscriber_bill
(
    p_subscriber_id IN NUMBER
)
AS
    v_total_bill NUMBER(10,2);
BEGIN
    SELECT NVL(SUM(charge_amount), 0)
    INTO v_total_bill
    FROM call_records
    WHERE subscriber_id = p_subscriber_id;

    DBMS_OUTPUT.PUT_LINE('Subscri  2  ber ID: ' || p_subscriber_id);
    DBMS_OUTPUT.PUT_LINE('Total Bill: ' || v_total_bill);
END;
